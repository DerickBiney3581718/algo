function partition(seq, kParts, grid) {
const seqLen = seq.length;

// 1. Pre-calculate prefix sums for speed
const prefixSum = [0];
for (let i = 0; i < seqLen; i++) prefixSum[i + 1] = prefixSum[i] + seq[i];

// 2. Fill the DP table
for (let i = 2; i <= seqLen; i++) { // For each number of elements
for (let k = 2; k <= kParts; k++) { // For each number of partitions
let bestMaxForThisCell = Infinity;

      // j is the index where the LAST partition starts (0-based)
      for (let j = 0; j < i; j++) {
        const leftSideCost = grid[j][k - 1]; // Best of k-1 parts for first j elements
        const rightSideCost = prefixSum[i] - prefixSum[j]; // Sum from j to i

        const currentMax = Math.max(leftSideCost, rightSideCost);

        if (currentMax < bestMaxForThisCell) {
          bestMaxForThisCell = currentMax;
        }
      }
      grid[i][k] = bestMaxForThisCell;
    }

}
}
